SELECT m.NAME
FROM MEAL m
LEFT JOIN INGREDIENT i
ON m.MEAL_ID = i.MEAL_ID;

SELECT m.NAME
FROM MEAL m
LEFT JOIN INGREDIENT i
ON m.MEAL_ID = i.MEAL_ID
WHERE i.MEAL_ID IS NULL;

SELECT m.NAME
FROM MEAL m
RIGHT JOIN INGREDIENT i
ON m.MEAL_ID = i.MEAL_ID;

SELECT m.NAME
FROM MEAL m
RIGHT JOIN INGREDIENT i
ON m.MEAL_ID = i.MEAL_ID;
WHERE m.MEAL_ID IS NULL

SELECT m.NAME
FROM MEAL m
FULL OUTER JOIN INGREDIENT i
ON m.MEAL_ID = i.MEAL_ID;

SELECT m.NAME
FROM MEAL m
FULL OUTER JOIN INGREDIENT i
ON m.MEAL_ID = i.MEAL_ID
WHERE m.MEAL_ID IS NULL
OR i.MEAL_ID IS NULL;
